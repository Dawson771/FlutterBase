支持设置多个定时任务，到点弹出悬浮窗 / 对话框  播放本地 / 网络视频，支持视频定点跳转



支持弹窗播放、悬浮窗播放、画中画模式，可通过 API 实现定点跳转，适合需要复杂视频播放功能的场景

## 功能拆解

定时触发：到指定时间唤醒应用、弹出界面（后台保活 + 定时任务是关键）

弹窗 / 悬浮窗：弹出独立界面播放视频（系统弹窗 / 悬浮窗权限）

定点播放：跳转视频指定时间点并自动播放（播放器 API 支持）

import 'package:flutter/material.dart';
import 'package:video_player/video_player.dart';
import 'package:flutter_local_notifications/flutter_local_notifications.dart';
import 'package:workmanager/workmanager.dart';

// 初始化通知插件
final FlutterLocalNotificationsPlugin flutterLocalNotificationsPlugin =
    FlutterLocalNotificationsPlugin();

void main() async {
  WidgetsFlutterBinding.ensureInitialized();

  // 1. 初始化本地通知
  const AndroidInitializationSettings androidSettings =
      AndroidInitializationSettings('@mipmap/ic_launcher');
  const InitializationSettings initializationSettings =
      InitializationSettings(android: androidSettings);
  await flutterLocalNotificationsPlugin.initialize(initializationSettings,
      onDidReceiveNotificationResponse: (response) {
    // 点击通知：打开视频播放弹窗
    if (response.payload != null) {
      runApp(const VideoPlayerApp(payload: 'alarm_video'));
    }
  });

  // 2. 初始化后台任务
  await Workmanager().initialize(
    callbackDispatcher,
    isInDebugMode: false,
  );

  // 3. 注册定时任务：5秒后触发（实际使用可设置具体时间）
  await Workmanager().registerOneOffTask(
    "video_alarm_task",
    "videoAlarm",
    initialDelay: const Duration(seconds: 5),
  );

  runApp(const VideoPlayerApp(payload: null));
}

// 后台任务回调
@pragma('vm:entry-point')
void callbackDispatcher() {
  Workmanager().executeTask((task, inputData) async {
    // 后台触发：发送本地通知
    await showVideoAlarmNotification();
    return Future.value(true);
  });
}

// 发送视频闹钟通知
Future<void> showVideoAlarmNotification() async {
  const AndroidNotificationDetails androidDetails = AndroidNotificationDetails(
    'video_alarm_channel',
    '视频闹钟频道',
    importance: Importance.max,
    priority: Priority.high,
    fullScreenIntent: true, // 全屏弹出（Android 10+ 支持）
  );
  const NotificationDetails notificationDetails =
      NotificationDetails(android: androidDetails);
  await flutterLocalNotificationsPlugin.show(
    0,
    '定点视频提醒',
    '到点了，开始播放视频',
    notificationDetails,
    payload: 'alarm_video', // 携带视频标识
  );
}

// 视频播放应用
class VideoPlayerApp extends StatelessWidget {
  final String? payload;

  const VideoPlayerApp({super.key, this.payload});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(title: const Text('定点视频播放')),
        body: Center(
          child: payload != null
              ? const VideoPlayerWidget() // 点击通知：显示视频播放
              : const Text('等待定时触发...'), // 初始界面
        ),
      ),
    );
  }
}

// 视频播放组件（含定点跳转）
class VideoPlayerWidget extends StatefulWidget {
  const VideoPlayerWidget({super.key});

  @override
  State<VideoPlayerWidget> createState() => _VideoPlayerWidgetState();
}

class _VideoPlayerWidgetState extends State<VideoPlayerWidget> {
  late VideoPlayerController _controller;
  late Future<void> _initializeVideoFuture;

  @override
  void initState() {
    super.initState();
    // 1. 初始化视频控制器（本地视频：需在 assets 中配置，或使用网络视频）
    _controller = VideoPlayerController.asset(
      'assets/videos/test.mp4', // 替换为你的视频路径
    );

    // 2. 初始化视频
    _initializeVideoFuture = _controller.initialize().then((_) {
      // 3. 定点播放：跳转到1分30秒（Duration 配置时间）
      _controller.seekTo(const Duration(minutes: 1, seconds: 30));
      // 4. 自动播放
      _controller.play();
      setState(() {});
    });
  }

  @override
  void dispose() {
    // 释放控制器资源
    _controller.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return FutureBuilder(
      future: _initializeVideoFuture,
      builder: (context, snapshot) {
        if (snapshot.connectionState == ConnectionState.done) {
          // 视频初始化完成：显示播放视图
          return AspectRatio(
            aspectRatio: _controller.value.aspectRatio,
            child: VideoPlayer(_controller),
          );
        } else {
          // 加载中
          return const CircularProgressIndicator();
        }
      },
    );
  }
}
