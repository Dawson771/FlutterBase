### `fluwx`（目前 Flutter 生态最成熟的微信相关插件）

`fluwx` 是 Flutter 生态中对接微信 SDK 最完善、维护最活跃的插件之一，支持**微信支付、微信分享、微信登录**三大核心功能，适配最新的微信 SDK 和 Flutter 版本，是集成微信支付的首选。

#### 核心优势

1. 双平台全适配（iOS + Android），无需自己手动封装原生代码，减少跨平台兼容问题。
2. 功能完整，支持微信支付的所有核心场景（APP 支付、扫码支付等，Flutter 项目主要用 APP 支付）。
3. 文档详细，社区案例丰富，遇到问题容易找到解决方案。
4. 支持后续回调处理、订单状态查询等关键流程，满足生产环境需求。

#### 基本信息

- 插件地址：https://pub.dev/packages/fluwx

- 前置准备（必须完成，否则集成会失败）

  1. 注册**微信开放平台**账号（https://open.weixin.qq.com/），完成开发者认证（企业 / 个体工商户，个人账号无法申请微信支付功能）。
  2. 创建移动应用，填写应用信息（iOS 的`Bundle ID`、Android 的`Application ID`），并上传应用签名（Android 需用微信提供的签名工具获取）。
  3. 申请微信支付功能，获取核心配置参数：`AppID`（微信应用 ID）、`MCH_ID`（商户号）、`API_KEY`（商户 API 密钥，用于签名生成）。
  4. iOS：配置 Xcode 项目的`URL Scheme`（为`wx${你的AppID}`，如`wx1234567890abcdef`），开启关联域名（如需）。
  5. Android：在`AndroidManifest.xml`中配置权限、注册微信回调 Activity，配置`AppID`。

### 完整集成步骤（核心流程）

#### 步骤 1：添加依赖

在`pubspec.yaml`中添加`fluwx`依赖，建议使用最新稳定版：

yaml











```
dependencies:
  flutter:
    sdk: flutter
  fluwx: ^3.15.0 # 查看pub.dev获取最新版本
```

添加后执行`flutter pub get`拉取依赖。

#### 步骤 2：双平台原生配置（关键，缺一不可）

##### Android 配置（`android`目录下）

1. 配置`AndroidManifest.xml`

- 添加权限（在`<manifest>`标签内，`<application>`标签外）：

xml











```
<!-- 网络权限（支付需要请求后端接口、与微信通信） -->
<uses-permission android:name="android.permission.INTERNET" />
<uses-permission android:name="android.permission.ACCESS_NETWORK_STATE" />
<uses-permission android:name="android.permission.ACCESS_WIFI_STATE" />
```

- 注册微信回调 Activity（在`<application>`标签内）：

xml











```
<!-- 微信支付回调Activity，包名必须与项目一致，且路径固定为wxapi/WXPayEntryActivity -->
<activity
    android:name=".wxapi.WXPayEntryActivity"
    android:exported="true"
    android:launchMode="singleTop">
    <intent-filter>
        <action android:name="android.intent.action.VIEW" />
        <category android:name="android.intent.category.DEFAULT" />
        <data android:scheme="wx${你的AppID}" /> <!-- 替换为你的微信AppID，如wx1234567890 -->
    </intent-filter>
</activity>
```

1. 创建回调 Activity 目录和文件

- 在`android/app/src/main/kotlin/你的项目包名/`下，创建`wxapi`目录（如`com/your/app/wxapi`）。
- 在`wxapi`目录下创建`WXPayEntryActivity.kt`（Kotlin）或`WXPayEntryActivity.java`（Java），`fluwx`会自动处理回调逻辑，只需简单实现：

kotlin











```
package com.your.app.wxapi // 替换为你的项目包名

import com.tencent.mm.opensdk.openapi.WXAPIFactory
import com.github.bingoohuang.fluwx.handlers.WXPayEntryActivityHandler

class WXPayEntryActivity : WXPayEntryActivityHandler() {
    override fun getWXApi() = WXAPIFactory.createWXAPI(this, "你的微信AppID", true)
}
```

##### iOS 配置（`ios`目录下）

1. 配置 URL Scheme

- 打开 Xcode（右键`ios/Runner.xcworkspace`用 Xcode 打开）。
- 选中`Runner` -> `Target` -> `Info` -> `URL Types`，点击`+`添加新的 URL Type。
- `Identifier`填写`weixin`，`URL Schemes`填写`wx${你的AppID}`（如`wx1234567890`）。

1. 配置回调处理

- 打开`ios/Runner/AppDelegate.swift`，添加`fluwx`的回调处理：

swift











```
import UIKit
import Flutter
import fluwx

@UIApplicationMain
@objc class AppDelegate: FlutterAppDelegate {
  override func application(
    _ application: UIApplication,
    didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?
  ) -> Bool {
    GeneratedPluginRegistrant.register(with: self)
    // 初始化fluwx
    FluwxPlugin.registerWxApi(appId: "你的微信AppID", universalLink: "你的Universal Link（可选，如需深度链接）")
    return super.application(application, didFinishLaunchingWithOptions: launchOptions)
  }
  
  // 处理微信回调URL
  override func application(_ app: UIApplication, open url: URL, options: [UIApplication.OpenURLOptionsKey : Any] = [:]) -> Bool {
      return FluwxPlugin.handleOpen(url, options: options)
  }
  
  // 处理Universal Link回调（可选）
  override func application(_ application: UIApplication, continue userActivity: NSUserActivity, restorationHandler: @escaping ([UIUserActivityRestoring]?) -> Void) -> Bool {
      return FluwxPlugin.handleContinue(userActivity, restorationHandler: restorationHandler)
  }
}
```

#### 步骤 3：Flutter 端核心代码（初始化 + 发起支付 + 监听回调）

微信支付的核心逻辑：**前端向后端请求支付参数 -> 后端调用微信支付接口生成预支付订单，返回核心参数 -> 前端用`fluwx`传入参数发起支付 -> 监听支付结果并同步到后端**。

> 注意：支付签名必须由后端生成，前端不能直接处理`API_KEY`，否则会造成密钥泄露，存在安全风险。

dart











```
import 'package:flutter/material.dart';
import 'package:fluwx/fluwx.dart' as fluwx;

class WxPayDemoPage extends StatefulWidget {
  const WxPayDemoPage({super.key});

  @override
  State<WxPayDemoPage> createState() => _WxPayDemoPageState();
}

class _WxPayDemoPageState extends State<WxPayDemoPage> {
  // 标记微信是否已安装
  bool _isWxInstalled = false;

  @override
  void initState() {
    super.initState();
    // 1. 初始化fluwx并检查微信安装状态
    _initWxPay();
    // 2. 监听微信支付结果回调
    _listenWxPayResult();
  }

  // 初始化微信支付
  Future<void> _initWxPay() async {
    // 注册微信AppID（与开放平台、原生配置中的AppID一致）
    await fluwx.registerWxApi(
      appId: "你的微信AppID", // 替换为你的真实AppID
      universalLink: "你的Universal Link（iOS可选）",
    );
    // 检查设备是否安装微信
    final bool isInstalled = await fluwx.isWeChatInstalled();
    if (!mounted) return;
    setState(() {
      _isWxInstalled = isInstalled;
    });
  }

  // 监听微信支付结果
  void _listenWxPayResult() {
    fluwx.weChatResponseStream.listen((res) {
      // 筛选支付结果回调
      if (res is fluwx.WeChatPayResponse) {
        switch (res.errCode) {
          case 0:
            // 支付成功（仅前端回调，最终需以后端接收到微信的异步通知为准）
            showDialog(
              context: context,
              builder: (context) => AlertDialog(title: const Text('支付成功'), actions: [TextButton(onPressed: () => Navigator.pop(context), child: const Text('确定'))]),
            );
            // 通知后端：查询订单最终状态（防止前端回调丢失）
            _notifyBackendQueryOrder();
            break;
          case -1:
            // 支付失败（参数错误、签名错误等）
            showDialog(
              context: context,
              builder: (context) => AlertDialog(title: const Text('支付失败'), content: Text(res.errStr ?? '未知错误'), actions: [TextButton(onPressed: () => Navigator.pop(context), child: const Text('确定'))]),
            );
            break;
          case -2:
            // 用户取消支付
            showDialog(
              context: context,
              builder: (context) => AlertDialog(title: const Text('已取消支付'), actions: [TextButton(onPressed: () => Navigator.pop(context), child: const Text('确定'))]),
            );
            break;
        }
      }
    });
  }

  // 发起微信支付（核心步骤）
  Future<void> _startWxPay() async {
    if (!_isWxInstalled) {
      showDialog(
        context: context,
        builder: (context) => AlertDialog(title: const Text('提示'), content: const Text('当前设备未安装微信'), actions: [TextButton(onPressed: () => Navigator.pop(context), child: const Text('确定'))]),
      );
      return;
    }

    try {
      // 步骤1：向后端请求微信支付预支付参数（必须由后端返回，以下是模拟数据，需替换为真实接口请求）
      final Map<String, dynamic> payParams = await _getPayParamsFromBackend();

      // 步骤2：构造微信支付请求参数
      final fluwx.WeChatPayRequest payRequest = fluwx.WeChatPayRequest(
        appId: payParams['appid'], // 微信AppID
        partnerId: payParams['partnerid'], // 商户号（MCH_ID）
        prepayId: payParams['prepayid'], // 预支付交易会话ID
        packageValue: payParams['package'], // 固定值：Sign=WXPay
        nonceStr: payParams['noncestr'], // 随机字符串
        timeStamp: payParams['timestamp'], // 时间戳（字符串格式）
        sign: payParams['sign'], // 支付签名（后端生成）
      );

      // 步骤3：发起微信支付
      await fluwx.sendWeChatPayRequest(payRequest);
    } catch (e) {
      showDialog(
        context: context,
        builder: (context) => AlertDialog(title: const Text('发起支付失败'), content: Text(e.toString()), actions: [TextButton(onPressed: () => Navigator.pop(context), child: const Text('确定'))]),
      );
    }
  }

  // 模拟向后端请求支付参数（真实项目中替换为你的后端接口）
  Future<Map<String, dynamic>> _getPayParamsFromBackend() async {
    // 这里仅做模拟，真实场景需用dio等网络库请求后端接口
    await Future.delayed(const Duration(milliseconds: 500));
    return {
      'appid': '你的微信AppID',
      'partnerid': '你的商户号MCH_ID',
      'prepayid': 'wx202601241234567890abcdefghijklmn', // 后端返回的预支付ID
      'package': 'Sign=WXPay', // 固定值，无需修改
      'noncestr': 'abcdef123456', // 后端生成的随机字符串
      'timestamp': '1737768900', // 后端生成的时间戳（字符串格式）
      'sign': 'XXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXX', // 后端生成的支付签名
    };
  }

  // 通知后端查询订单最终状态
  Future<void> _notifyBackendQueryOrder() async {
    // 真实项目中：传入订单号，请求后端接口查询微信订单最终状态
    debugPrint('通知后端查询订单状态');
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('微信支付示例')),
      body: Center(
        child: ElevatedButton(
          onPressed: _startWxPay,
          child: const Text('发起微信支付'),
        ),
      ),
    );
  }
}
```

#### 步骤 4：关键补充（生产环境注意事项）

1. 后端必须对接微信支付的**异步通知接口**（微信支付成功后，会主动回调后端配置的通知地址），订单的最终状态以异步通知为准，而非前端回调（前端回调可能丢失、被篡改）。
2. Android 签名问题：正式打包时，必须使用与微信开放平台备案一致的签名文件（`.jks`），否则会提示 “签名错误” 或 “应用未注册”。
3. iOS 审核：微信支付集成后，App Store 审核时需提供相关资质（微信支付授权文件），否则可能审核不通过。
4. 异常处理：需处理 “微信未安装”“网络异常”“预支付订单过期” 等场景，提升用户体验。
